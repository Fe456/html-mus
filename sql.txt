CREATE DATABASE WEB;
USE WEB;

DROP TABLE IF EXISTS usuario;
CREATE TABLE USUARIO (
ID INT auto_increment PRIMARY KEY,
EMAIL VARCHAR(255),
SENHA VARCHAR(255),
STATUS bool);

DROP TABLE IF EXISTS musicas;
CREATE TABLE musicas (
ID INT auto_increment primary key,
NOME VARCHAR(255),
ARTISTA VARCHAR(255),
IMG_COVER MEDIUMBLOB,
AUDIO LONGBLOB,
AUDIOTYPE VARCHAR(50),
DURATION INT);

SELECT * FROM USUARIO;
SELECT * FROM MUSICAS;
DELETE FROM USUARIO;
ALTER TABLE USUARIO
AUTO_INCREMENT = 0;
DELETE FROM MUSICAS;
ALTER TABLE MUSICAS
AUTO_INCREMENT = 0;
